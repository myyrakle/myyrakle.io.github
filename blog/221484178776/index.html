[node.js] 'Hello World' 서버 돌려보기

  • 2019.03.09
  • ::
  • 1 min read
#Javascript
[원본 링크]

코드는 이렇게 짜면 된다.

const http = require('http');

//사용할 포트 번호(임의)
const port = 52273;

<br>

function html_writer(request, response)
{
    //헤더 써줌
    response.writeHead(200/*성공을 뜻함*/, {'Content-Type':'text/html'});

    //본문 써줌
    response.end('<h1>Hello World!!</h1>')
}

//서버 생성
const server = http.createServer(html_writer);

function log_writer()
{   
    //실행중에 계속 호출
    console.log('running');
}    
//서버 실행
server.listen(port, log_writer);

그리고 실행을 시킨다.


그리고 브라우저를 아무나 켜서 주소창에
http://127.0.0.1:52273/
나
http://localhost:52273/
로 들어가서


이렇게 뜨면 성공이다.

다만 이 방법은 서버 pc에서만 사용할 수 있는 방법이고,
다른 데서 접근하려면 저 아이피 칸에다 pc의 아이피를 적어주면 된다.

아이피를 알아내는 방법은 여러가지가 있는데.. 그냥 네이버에 쳐도 나온다.

그리고 이걸 주소 부분에 그대로 넣어주면 된다.

폰을 켜서 확인해봐도 좋다.
