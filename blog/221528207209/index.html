ES6부터 클래스 문법이 생겼다.

근데 뭐 사실 대단한 기능이 추가된건 아니고, 단순히 편의성과 가독성을 위해 추가된거라고 한다.

대충 이렇게 생겼다.

class Person {
    //생성자다.
    constructor(name, age){
        this._name = name;
        this._age = age
    }

    //읽기전용 프로퍼티다.
    get Name() {
        return this._name;
    }
    //쓰기전용 프로퍼티다.
    set Name(name) {
        this._name = name;
    }

    get Age(){  
        return this._age; 
    }

    //그냥 메서드다.
    AgeInc(){
        this._age++;
    }
}

var john = new Person('john', '3000');
john.Name='tom'; 
john.AgeInc()
console.log(john.Name, john.Age);

상속도 좀더 보기 좋게 제공된다.
접근제어는 안 되는 것 같다만..

//Person을 상속받는 Student 클래스
class Student extends Person
{
    constructor(name, age, school){
        super(name, age); //상위 클래스 초기화
        this._school = school;
    }

    get School() {
        return this._school; 
    }
}

var john = new Student('john', '3000', '호그와트');
console.log(john.Name, john.Age, john.School);
