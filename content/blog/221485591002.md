```javascript
const http = require('http');

//사용할 포트 번호(임의)
const port = 52273;

//서버 생성
const server = http.createServer(
    function(request, response)
    {
        response.writeHead(200, {'Content-Type':'text/html'})
        response.end('<h2>Hello World!</h2>')
    }
);
//이벤트 세팅
server.on('request', function(){console.log('request 발생')})
server.on('connection', function(){console.log('클라이언트 접속')})
server.on('close', function(){console.log('서버 종료')})
server.on('clientError', function(){console.log('클라이언트 에러')})

//서버 실행
server.listen(port, function(){ console.log('running') });
```
